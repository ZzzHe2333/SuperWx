- AutomationId: session_list
- ClassName: mmui::XTableView

修改 ChatBox.init()：

要求：
1. 新 locator 优先。
2. 原有旧定位逻辑必须保留为 fallback。
3. 不要用输入框 Name 定位。
4. send_button 优先用 Name="发送" + ClassName="mmui::XOutlineButton"。
5. 如果 send_button 找不到，不要直接报错，因为输入框为空时发送按钮可能不存在。
6. msgbox 和 editbox 必须尽可能定位成功。

推荐逻辑：

def init(self):
    chat_page = find_first(self.control, SELECTORS["chat_message_page"], search_depth=30)
    root = chat_page if ctrl_exists(chat_page) else self.control

    message_view = find_first(root, SELECTORS["message_view"], search_depth=30)

    self.msgbox = find_first(root, SELECTORS["chat_message_list"], search_depth=30)
    if not ctrl_exists(self.msgbox) and ctrl_exists(message_view):
        self.msgbox = find_first(message_view, SELECTORS["chat_message_list"], search_depth=30)

    if not ctrl_exists(self.msgbox):
        self.msgbox = (
            self.control.ListControl(AutomationId="chat_message_list", ClassName="mmui::RecyclerListView")
            or self.control.ListControl(AutomationId="chat_message_list")
            or self.control.ListControl(ClassName="mmui::RecyclerListView")
            or self.control.GroupControl(ClassName="mmui::MessageView").ListControl(AutomationId="chat_message_list")
            or self.control.GroupControl(ClassName="mmui::MessageView").ListControl()
            or self.control.ListControl()
        )

    self.input_view = find_first(root, SELECTORS["input_view"], search_depth=30)

    self.editbox = find_first(root, SELECTORS["chat_input_field"], search_depth=30)
    if not ctrl_exists(self.editbox) and ctrl_exists(self.input_view):
        self.editbox = find_first(self.input_view, SELECTORS["chat_input_field"], search_depth=30)

    if not ctrl_exists(self.editbox):
        self.editbox = (
            self.control.EditControl(AutomationId="chat_input_field", ClassName="mmui::ChatInputField")
            or self.control.EditControl(AutomationId="chat_input_field")
            or self.control.EditControl(ClassName="mmui::ChatInputField")
        )

    self.sendbtn = find_first(root, SELECTORS["send_button"], search_depth=30)
    if not ctrl_exists(self.sendbtn) and ctrl_exists(self.input_view):
        self.sendbtn = find_first(self.input_view, SELECTORS["send_button"], search_depth=30)

    if not ctrl_exists(self.sendbtn):
        self.sendbtn = (
            self.control.ButtonControl(Name="发送", ClassName="mmui::XOutlineButton")
            or self.control.ButtonControl(Name="发送")
            or self.control.ButtonControl(Name="发送(S)")
            or self.control.ButtonControl(Name="Send")
        )

    self.tools = (
        self.control.ToolBarControl(AutomationId="tool_bar_accessible")
        or self.control.ToolBarControl()
    )

    self._empty = False

    if (cid := self.id) and cid not in USED_MSG_IDS:
        if ctrl_exists(self.msgbox):
            USED_MSG_IDS[self.id] = tuple((i.runtimeid for i in self.msgbox.GetChildren()))
        else:
            USED_MSG_IDS[self.id] = tuple()

    if not USED_MSG_IDS.get(cid):
        self._empty = True
